```python
from hypothesis import given, strategies as st
from datetime import date, datetime, timedelta
from io import StringIO

from dateutil import parser, tz

# Summary: Generate only ISO-8601 strings documented as valid, covering common dates
# YYYY / YYYY-MM / YYYY-MM-DD, basic and extended forms, ISO week dates with and
# without explicit weekday, optional times in all documented precisions, fractional
# seconds with "." or ",", 24:00 midnight rollover, optional UTC and numeric offsets,
# and both string and stream inputs. The property checks that parsed components equal
# the documented defaults/values, that 24:00 rolls to the next day, and that UTC-like
# offsets are represented as UTC. It avoids asserting rejection of invalid strings
# because the documentation warns parser strictness is not stable.
@given(st.data())
def test_dateutil_parser_isoparse(data):
    year_strategy = st.one_of(
        st.sampled_from([1, 4, 99, 999, 1000, 1582, 1900, 1999, 2000, 2020, 9999]),
        st.integers(min_value=1, max_value=9999),
    )
    date_strategy = st.one_of(
        st.sampled_from(
            [
                date(1, 1, 1),
                date(1999, 12, 31),
                date(2000, 2, 29),
                date(2020, 2, 29),
                date(9999, 12, 30),  # keep one day available for 24:00 rollover
            ]
        ),
        st.dates(min_value=date(1, 1, 1), max_value=date(9999, 12, 30)),
    )

    date_kind = data.draw(
        st.sampled_from(["year", "month", "calendar", "week", "week_day"])
    )

    complete_date = False

    if date_kind == "year":
        y = data.draw(year_strategy)
        date_part = f"{y:04d}"
        expected_date = date(y, 1, 1)

    elif date_kind == "month":
        y = data.draw(year_strategy)
        m = data.draw(st.integers(min_value=1, max_value=12))
        extended = data.draw(st.booleans())
        date_part = f"{y:04d}-{m:02d}" if extended else f"{y:04d}{m:02d}"
        expected_date = date(y, m, 1)

    elif date_kind == "calendar":
        expected_date = data.draw(date_strategy)
        extended = data.draw(st.booleans())
        if extended:
            date_part = (
                f"{expected_date.year:04d}-"
                f"{expected_date.month:02d}-"
                f"{expected_date.day:02d}"
            )
        else:
            date_part = (
                f"{expected_date.year:04d}"
                f"{expected_date.month:02d}"
                f"{expected_date.day:02d}"
            )
        complete_date = True

    elif date_kind == "week":
        source_date = data.draw(date_strategy)
        iso_year, iso_week, _ = source_date.isocalendar()
        extended = data.draw(st.booleans())
        date_part = (
            f"{iso_year:04d}-W{iso_week:02d}"
            if extended
            else f"{iso_year:04d}W{iso_week:02d}"
        )
        expected_date = date.fromisocalendar(iso_year, iso_week, 1)

    else:
        source_date = data.draw(date_strategy)
        iso_year, iso_week, iso_day = source_date.isocalendar()
        extended = data.draw(st.booleans())
        date_part = (
            f"{iso_year:04d}-W{iso_week:02d}-{iso_day}"
            if extended
            else f"{iso_year:04d}W{iso_week:02d}{iso_day}"
        )
        expected_date = date.fromisocalendar(iso_year, iso_week, iso_day)
        complete_date = True

    include_time = complete_date and data.draw(st.booleans())

    hour = minute = second = microsecond = 0
    tzinfo = None
    time_part = ""

    if include_time:
        use_24_hour_midnight = data.draw(st.booleans())

        if use_24_hour_midnight:
            hour = 24
            minute = second = microsecond = 0
            time_kind = data.draw(st.sampled_from(["hm", "hms", "fraction"]))
        else:
            hour = data.draw(st.integers(min_value=0, max_value=23))
            time_kind = data.draw(st.sampled_from(["h", "hm", "hms", "fraction"]))

        if time_kind == "h":
            time_body = f"{hour:02d}"

        elif time_kind == "hm":
            if hour != 24:
                minute = data.draw(st.integers(min_value=0, max_value=59))
            extended = data.draw(st.booleans())
            time_body = (
                f"{hour:02d}:{minute:02d}"
                if extended
                else f"{hour:02d}{minute:02d}"
            )

        elif time_kind == "hms":
            if hour != 24:
                minute = data.draw(st.integers(min_value=0, max_value=59))
                second = data.draw(st.integers(min_value=0, max_value=59))
            extended = data.draw(st.booleans())
            time_body = (
                f"{hour:02d}:{minute:02d}:{second:02d}"
                if extended
                else f"{hour:02d}{minute:02d}{second:02d}"
            )

        else:
            if hour != 24:
                minute = data.draw(st.integers(min_value=0, max_value=59))
                second = data.draw(st.integers(min_value=0, max_value=59))

            frac_len = data.draw(st.integers(min_value=1, max_value=6))
            if hour == 24:
                frac_digits = "0" * frac_len
            else:
                frac_digits = data.draw(
                    st.text(
                        alphabet="0123456789",
                        min_size=frac_len,
                        max_size=frac_len,
                    )
                )
            microsecond = int(frac_digits.ljust(6, "0"))
            decimal_separator = data.draw(st.sampled_from([".", ","]))
            time_body = (
                f"{hour:02d}:{minute:02d}:{second:02d}"
                f"{decimal_separator}{frac_digits}"
            )

        offset = None
        tz_kind = data.draw(st.sampled_from(["none", "Z", "offset"]))

        if tz_kind == "Z":
            tz_part = "Z"
            offset = timedelta(0)
            tzinfo = tz.tzutc()

        elif tz_kind == "offset":
            offset_format = data.draw(st.sampled_from(["HH", "HHMM", "HH:MM"]))
            sign_text = data.draw(st.sampled_from(["+", "-"]))
            sign = 1 if sign_text == "+" else -1
            offset_hour = data.draw(st.integers(min_value=0, max_value=23))
            offset_minute = (
                0
                if offset_format == "HH"
                else data.draw(st.integers(min_value=0, max_value=59))
            )

            if offset_format == "HH":
                tz_part = f"{sign_text}{offset_hour:02d}"
            elif offset_format == "HHMM":
                tz_part = f"{sign_text}{offset_hour:02d}{offset_minute:02d}"
            else:
                tz_part = f"{sign_text}{offset_hour:02d}:{offset_minute:02d}"

            offset = timedelta(
                seconds=sign * (offset_hour * 3600 + offset_minute * 60)
            )
            tzinfo = tz.tzutc() if offset == timedelta(0) else tz.tzoffset(None, offset)

        else:
            tz_part = ""

        separator = data.draw(st.sampled_from(["T", " ", "_"]))
        time_part = separator + time_body + tz_part

    dt_str = date_part + time_part

    if include_time and hour == 24:
        expected_date = expected_date + timedelta(days=1)
        hour = 0

    expected = datetime(
        expected_date.year,
        expected_date.month,
        expected_date.day,
        hour,
        minute,
        second,
        microsecond,
        tzinfo=tzinfo,
    )

    input_value = data.draw(st.sampled_from([dt_str, StringIO(dt_str)]))
    parsed = parser.isoparse(input_value)

    assert isinstance(parsed, datetime)
    assert parsed == expected
    assert parsed.year == expected.year
    assert parsed.month == expected.month
    assert parsed.day == expected.day
    assert parsed.hour == expected.hour
    assert parsed.minute == expected.minute
    assert parsed.second == expected.second
    assert parsed.microsecond == expected.microsecond
    assert parsed.utcoffset() == expected.utcoffset()

    if expected.tzinfo is None:
        assert parsed.tzinfo is None
    elif expected.utcoffset() == timedelta(0):
        assert parsed.tzinfo == tz.tzutc()
# End program
```
