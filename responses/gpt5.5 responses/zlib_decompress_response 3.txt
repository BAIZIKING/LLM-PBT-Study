from hypothesis import given, strategies as st
import zlib

# Summary: Strategy: generate both valid compressed streams and arbitrary malformed byte strings. Valid streams are built from edge-heavy/random payloads, varied compression levels, zlib/raw/gzip wrappers, different window sizes, compatible decompression wbits values including auto-detect modes, and edge/random bufsize values. Properties: valid streams round-trip to the original bytes, bufsize does not affect the decompressed output, and arbitrary inputs either return deterministic bytes or raise zlib.error as documented.
@given(st.data())
def test_zlib_decompress(data):
    payload_strategy = st.one_of(
        st.binary(min_size=0, max_size=8192),
        st.sampled_from([
            b"",
            b"\x00",
            b"\xff",
            b"a",
            b"hello world",
            b"\x00" * 1024,
            b"a" * 32768,
            bytes(range(256)),
        ]),
        st.builds(
            lambda chunk, count: chunk * count,
            st.binary(min_size=1, max_size=16),
            st.integers(min_value=0, max_value=2048),
        ),
    )

    bufsize_strategy = st.one_of(
        st.sampled_from([0, 1, 2, 8, 16, 1024, 16384, zlib.DEF_BUF_SIZE]),
        st.integers(min_value=0, max_value=65536),
    )

    case = data.draw(st.sampled_from(["valid_stream", "arbitrary_bytes"]))

    if case == "valid_stream":
        payload = data.draw(payload_strategy)
        wrapper = data.draw(st.sampled_from(["zlib", "raw", "gzip"]))
        window_bits = data.draw(st.integers(min_value=9, max_value=15))
        level = data.draw(st.integers(min_value=-1, max_value=9))
        bufsize = data.draw(bufsize_strategy)
        other_bufsize = data.draw(bufsize_strategy)

        if wrapper == "zlib":
            compress_wbits = window_bits
            valid_decompress_wbits = (
                [0]
                + list(range(window_bits, 16))
                + list(range(32 + window_bits, 48))
            )
        elif wrapper == "raw":
            compress_wbits = -window_bits
            valid_decompress_wbits = list(range(-window_bits, -16, -1))
        else:
            compress_wbits = 16 + window_bits
            valid_decompress_wbits = (
                list(range(16 + window_bits, 32))
                + list(range(32 + window_bits, 48))
            )

        compressor = zlib.compressobj(level=level, wbits=compress_wbits)
        compressed = compressor.compress(payload) + compressor.flush()

        decompress_wbits = data.draw(st.sampled_from(valid_decompress_wbits))

        result = zlib.decompress(
            compressed,
            wbits=decompress_wbits,
            bufsize=bufsize,
        )

        assert isinstance(result, bytes)
        assert result == payload
        assert zlib.decompress(
            compressed,
            wbits=decompress_wbits,
            bufsize=other_bufsize,
        ) == result

    else:
        arbitrary_data = data.draw(st.binary(min_size=0, max_size=8192))
        wbits = data.draw(st.integers(min_value=-50, max_value=60))
        bufsize = data.draw(bufsize_strategy)
        other_bufsize = data.draw(bufsize_strategy)

        try:
            result = zlib.decompress(
                arbitrary_data,
                wbits=wbits,
                bufsize=bufsize,
            )
        except zlib.error:
            return

        assert isinstance(result, bytes)
        assert zlib.decompress(
            arbitrary_data,
            wbits=wbits,
            bufsize=other_bufsize,
        ) == result
# End program
