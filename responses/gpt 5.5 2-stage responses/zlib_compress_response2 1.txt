from hypothesis import given, strategies as st
import zlib


@given(st.data())
def test_zlib_compress_returns_bytes(data):
    input_data = data.draw(st.binary(max_size=100_000))
    level = data.draw(st.integers(min_value=-1, max_value=9))
    wbits = data.draw(
        st.one_of(
            st.integers(min_value=9, max_value=15),
            st.integers(min_value=-15, max_value=-9),
            st.integers(min_value=25, max_value=31),
        )
    )

    compressed = zlib.compress(input_data, level=level, wbits=wbits)

    assert isinstance(compressed, bytes)


@given(st.data())
def test_zlib_compress_round_trips_with_matching_wbits(data):
    input_data = data.draw(st.binary(max_size=100_000))
    level = data.draw(st.integers(min_value=-1, max_value=9))
    wbits = data.draw(
        st.one_of(
            st.integers(min_value=9, max_value=15),
            st.integers(min_value=-15, max_value=-9),
            st.integers(min_value=25, max_value=31),
        )
    )

    compressed = zlib.compress(input_data, level=level, wbits=wbits)
    decompressed = zlib.decompress(compressed, wbits=wbits)

    assert decompressed == input_data


@given(st.data())
def test_zlib_compress_positive_wbits_has_valid_zlib_wrapper(data):
    input_data = data.draw(st.binary(max_size=100_000))
    level = data.draw(st.integers(min_value=-1, max_value=9))
    wbits = data.draw(st.integers(min_value=9, max_value=15))

    compressed = zlib.compress(input_data, level=level, wbits=wbits)

    assert len(compressed) >= 6
    assert compressed[0] & 0x0F == 8
    assert (((compressed[0] << 8) + compressed[1]) % 31) == 0
    assert compressed[1] & 0x20 == 0
    assert int.from_bytes(compressed[-4:], "big") == zlib.adler32(input_data) & 0xFFFFFFFF
    assert zlib.decompress(compressed, wbits=wbits) == input_data


@given(st.data())
def test_zlib_compress_negative_wbits_is_raw_deflate_without_zlib_wrapper(data):
    input_data = data.draw(st.binary(max_size=100_000))
    level = data.draw(st.integers(min_value=-1, max_value=9))
    wbits = data.draw(st.integers(min_value=-15, max_value=-9))

    raw_compressed = zlib.compress(input_data, level=level, wbits=wbits)
    zlib_wrapped = zlib.compress(input_data, level=level, wbits=abs(wbits))

    assert zlib.decompress(raw_compressed, wbits=wbits) == input_data
    assert raw_compressed == zlib_wrapped[2:-4]


@given(st.data())
def test_zlib_compress_gzip_wbits_has_valid_gzip_wrapper(data):
    input_data = data.draw(st.binary(max_size=100_000))
    level = data.draw(st.integers(min_value=-1, max_value=9))
    wbits = data.draw(st.integers(min_value=25, max_value=31))

    compressed = zlib.compress(input_data, level=level, wbits=wbits)

    assert len(compressed) >= 18
    assert compressed[0:2] == b"\x1f\x8b"
    assert compressed[2] == 8
    assert compressed[3] == 0
    assert int.from_bytes(compressed[-8:-4], "little") == zlib.crc32(input_data) & 0xFFFFFFFF
    assert int.from_bytes(compressed[-4:], "little") == len(input_data) & 0xFFFFFFFF
    assert zlib.decompress(compressed, wbits=wbits) == input_data
