from hypothesis import given, strategies as st
import html

MAX_SIZE = 500

@given(st.data())
def test_html_escape_output_contains_no_raw_angle_brackets(data):
    s = data.draw(st.text(max_size=MAX_SIZE))
    quote = data.draw(st.booleans())

    escaped = html.escape(s, quote=quote)

    assert "<" not in escaped
    assert ">" not in escaped


@given(st.data())
def test_html_escape_ampersands_are_escaped_as_amp(data):
    s = data.draw(st.text(max_size=MAX_SIZE))
    quote = data.draw(st.booleans())

    escaped = html.escape(s, quote=quote)

    assert escaped.count("&amp;") == s.count("&")


@given(st.data())
def test_html_escape_quote_true_escapes_quotes(data):
    s = data.draw(st.text(max_size=MAX_SIZE))

    escaped = html.escape(s, quote=True)

    assert '"' not in escaped
    assert "'" not in escaped
    assert escaped.count("&quot;") == s.count('"')
    assert escaped.count("&#x27;") == s.count("'")


@given(st.data())
def test_html_escape_quote_false_preserves_quotes(data):
    s = data.draw(st.text(max_size=MAX_SIZE))

    escaped = html.escape(s, quote=False)

    input_quotes = "".join(ch for ch in s if ch in "\"'")
    output_quotes = "".join(ch for ch in escaped if ch in "\"'")

    assert output_quotes == input_quotes


@given(st.data())
def test_html_escape_preserves_safe_characters(data):
    quote = data.draw(st.booleans())
    escaped_characters = "&<>" + ("\"'" if quote else "")
    s = data.draw(
        st.text(
            alphabet=st.characters(blacklist_characters=escaped_characters),
            max_size=MAX_SIZE,
        )
    )

    escaped = html.escape(s, quote=quote)

    assert escaped == s
