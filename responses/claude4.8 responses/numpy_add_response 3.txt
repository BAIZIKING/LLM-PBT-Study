# Generation Strategy

To generate a wide variety of inputs and edge cases for `numpy.add`, I'll use the following approach:

1. **Diverse dtypes**: Test across integer types (int8, int16, int32, int64) and floating-point types (float16, float32, float64). Different dtypes expose different overflow/precision behaviors.

2. **Broadcastable shapes**: Use hypothesis's `numpy` extension to generate arrays with broadcastable shapes via `mutually_broadcastable_shapes`. This covers the documented broadcasting behavior, including scalar-like shapes `()`.

3. **Element value ranges**: Constrain element values to avoid overflow during property checks (which would make commutativity/associativity checks meaningless), but still include zeros, negatives, and boundary values.

4. **Edge cases**: Include empty arrays (via shapes with `0` dimensions), 0-dimensional arrays (scalars), and varied dimensionality (0 to several dimensions).

# Properties to Check

Based on the API documentation:

1. **Equivalence to `+` operator**: The docs state "The + operator can be used as a shorthand for np.add". So `np.add(x1, x2) == x1 + x2`.

2. **Commutativity**: Addition is commutative: `np.add(x1, x2) == np.add(x2, x1)`.

3. **Output shape equals broadcast shape**: The result shape must equal `np.broadcast_shapes(x1.shape, x2.shape)`.

4. **`out` parameter**: When `out` is provided, the function writes into it and returns the same object.

5. **`where` parameter**: At `False` locations, `out` retains its original value; at `True` locations it holds the sum.

6. **Identity element**: Adding zero returns an array equal to the original (broadcast).

@given(st.data())
def test_numpy_add_property(data):
    # --- choose a dtype ---
    dtype = data.draw(
        st.sampled_from([
            np.int8, np.int16, np.int32, np.int64,
            np.float16, np.float32, np.float64,
        ]),
        label="dtype",
    )

    # bounded element values to avoid overflow obscuring the algebraic properties
    if np.issubdtype(dtype, np.integer):
        elements = st.integers(min_value=-100, max_value=100)
    else:
        elements = st.floats(
            min_value=-1e3, max_value=1e3,
            allow_nan=False, allow_infinity=False, width=32,
        )

    # --- generate two broadcastable shapes ---
    shapes = data.draw(
        hnp.mutually_broadcastable_shapes(num_shapes=2, max_dims=4, max_side=4),
        label="shapes",
    )
    shape1, shape2 = shapes.input_shapes
    broadcast_shape = shapes.result_shape

    x1 = data.draw(hnp.arrays(dtype=dtype, shape=shape1, elements=elements), label="x1")
    x2 = data.draw(hnp.arrays(dtype=dtype, shape=shape2, elements=elements), label="x2")

    result = np.add(x1, x2)

    # Property 1: output shape equals the broadcast shape
    assert result.shape == broadcast_shape

    # Property 2: equivalence to the '+' operator
    expected = x1 + x2
    np.testing.assert_array_equal(result, expected)

    # Property 3: commutativity  np.add(x1, x2) == np.add(x2, x1)
    np.testing.assert_array_equal(result, np.add(x2, x1))

    # Property 4: 'out' parameter is written in-place and returned
    out = np.empty(broadcast_shape, dtype=dtype)
    returned = np.add(x1, x2, out=out)
    assert returned is out
    np.testing.assert_array_equal(out, expected)

    # Property 5: 'where' masking semantics
    sentinel = np.zeros(broadcast_shape, dtype=dtype)
    out2 = np.full(broadcast_shape, dtype(0), dtype=dtype)
    where_mask = data.draw(
        hnp.arrays(dtype=bool, shape=broadcast_shape), label="where",
    )
    np.add(x1, x2, out=out2, where=where_mask)
    # at True locations -> the sum; at False locations -> retains original (0)
    expected_masked = np.where(where_mask, expected, sentinel)
    np.testing.assert_array_equal(out2, expected_masked)

    # Property 6: zero is the additive identity (broadcasts to result shape)
    zero = np.zeros_like(x1)
    np.testing.assert_array_equal(np.add(x1, zero), x1 + zero)
    np.testing.assert_array_equal(np.add(x1, zero), np.broadcast_to(x1, x1.shape))
